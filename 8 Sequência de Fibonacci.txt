using System;
					
public class Program
{
	public static void Main()
	{
		int num1=0, num2=1, aux;
		int N=0;
		
		Console.WriteLine("Informe o número de componentes desejados da sequência de Fibonacci");
		N = Convert.ToInt32(Console.ReadLine());
		Console.WriteLine("A sequência é:");
		Console.WriteLine("1");
		
		for (int i = 0; i < (N-1); i++)
		{
			aux=num1;
			num1 = num2;
			num2 = num1 + aux;
			
			Console.WriteLine(num2);

		}
	}
}
